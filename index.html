<script>

    const img = document.getElementById('baseImg');

    const cvs = document.getElementById('ovl');

    const ctx = cvs.getContext('2d');


    function loadDemo() {

        document.getElementById('placeholderText').style.display = 'none';

        img.src = "https://images.unsplash.com/photo-1517336712642-19e09961446a?w=800";

        document.getElementById('qwenInput').value = '<ref>Ordinateur Portable</ref><box>(280,250,720,750)</box>\n<ref>Tasse à café</ref><polygon>(650,100,650,220,850,220,850,100)</polygon>';

        img.onload = () => { syncSize(); draw(); };

    }


    document.getElementById('imgLoad').onchange = (e) => {

        const reader = new FileReader();

        reader.onload = (ev) => { 

            img.src = ev.target.result; 

            document.getElementById('placeholderText').style.display = 'none';

        };

        reader.readAsDataURL(e.target.files[0]);

    };


    img.onload = syncSize;


    function syncSize() {

        cvs.width = img.clientWidth;

        cvs.height = img.clientHeight;

        cvs.style.width = img.clientWidth + "px";

        cvs.style.height = img.clientHeight + "px";

    }


    function draw() {

        const text = document.getElementById('qwenInput').value;

        const w = cvs.width; const h = cvs.height;

        ctx.clearRect(0, 0, w, h);

        

        const colors = ['rgba(255,50,50,0.4)', 'rgba(50,255,50,0.4)', 'rgba(50,100,255,0.4)'];

        const strokes = ['#ff3232', '#32ff32', '#3264ff'];


        const reg = /<ref>(.*?)<\/ref>\s*<(box|polygon)>\((.*?)\)<\/\2>/g;

        let m, i=0;

        while ((m = reg.exec(text)) !== null) {

            const label = m[1];

            const type = m[2];

            const coords = m[3].split(',').map(Number);

            const color = colors[i % 3];

            const stroke = strokes[i % 3];


            ctx.beginPath();

            if(type === 'box') {

                const [y1, x1, y2, x2] = coords.map((v,k) => k%2===0 ? v*h/1000 : v*w/1000);

                ctx.rect(x1, y1, x2-x1, y2-y1);

                labelAt(label, x1, y1, stroke);

            } else {

                ctx.moveTo(coords[1]*w/1000, coords[0]*h/1000);

                for(let j=0; j<coords.length; j+=2) ctx.lineTo(coords[j+1]*w/1000, coords[j]*h/1000);

                ctx.closePath();

                labelAt(label, coords[1]*w/1000, coords[0]*h/1000, stroke);

            }

            ctx.fillStyle = color; ctx.fill();

            ctx.strokeStyle = stroke; ctx.lineWidth = 3; ctx.stroke();

            i++;

        }

        document.getElementById('dlBtn').style.display = 'block';

    }


    function labelAt(txt, x, y, c) {

        ctx.fillStyle = c; ctx.fillRect(x, y-25, ctx.measureText(txt).width+15, 25);

        ctx.fillStyle = "white"; ctx.font = "bold 14px sans-serif";

        ctx.fillText(txt, x+5, y-8);

    }


    function download() {

        const finalCvs = document.createElement('canvas');

        finalCvs.width = img.naturalWidth; finalCvs.height = img.naturalHeight;

        const fctx = finalCvs.getContext('2d');

        fctx.drawImage(img, 0, 0);

        const link = document.createElement('a');

        link.download = 'qwen_result.png';

        link.href = cvs.toDataURL();

        link.click();

    }
